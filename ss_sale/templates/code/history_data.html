所有使用time均为unix时间，请自行转换。 

			
			
golang-获取全部历史日数据,示例代码

			
			
package main
import "fmt"
import "net/http"
func main(){
	resp,_ := http.Get("http://www.ccdata.tk/history_data")
	defer resp.Body.Close()
	body,_ := ioutil.ReadAll(resp.Body)
	fmt.Println(string(body))
}
			

			
golang-获取指定单日数据,示例代码

			
			
package main

golang-获取指定时间段日数据,示例代码

			
			
package main
import "fmt"

func main(){
	resp,_ := http.Get("http://www.ccdata.tk/history_data/{start_time}/{end_time}")

}
			

			


			
python-获取全部日数据,示例代码

			
建议使用第三方包request,比较简易.

		
import requests
import json
def get_json():
	r = requests.get('http://www.ccdata.tk/history_data')
	my_json = json.loads(r.text)
	print(my_json)

if __name__ == '__main__':
	get_json()
